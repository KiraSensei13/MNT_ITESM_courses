#eq 21
def K_(T, Delta_H_v_ref, T_ref, Delta_Cp):
    R = 8.314/1000;
    
    pwr1 = -(Delta_H_v_ref/R)*((1/T)-(1/T_ref));
    pwr2 = (Delta_Cp/R) * (np.log(T/T_ref) + (T_ref/T) - 1);
    
    res = np.exp(pwr1 + pwr2);
    return res

#eq 38
def alpha_1_(T, Delta_H_v_ref_1, Delta_H_v_ref_2, Delta_H_v_ref_3, T_ref_1, T_ref_2, T_ref_3, Delta_Cp_1, Delta_Cp_2, Delta_Cp_3):
    K_1 = K_(T, Delta_H_v_ref_1, T_ref_1, Delta_Cp_1)
    K_2 = K_(T, Delta_H_v_ref_2, T_ref_2, Delta_Cp_2)
    K_3 = K_(T, Delta_H_v_ref_3, T_ref_3, Delta_Cp_3)
    
    return (K_1)/(1 + K_1 + K_1*K_2 + K_1*K_2*K_3);

def alpha_2_(T, Delta_H_v_ref_1, Delta_H_v_ref_2, Delta_H_v_ref_3, T_ref_1, T_ref_2, T_ref_3, Delta_Cp_1, Delta_Cp_2, Delta_Cp_3):
    K_1 = K_(T, Delta_H_v_ref_1, T_ref_1, Delta_Cp_1)
    K_2 = K_(T, Delta_H_v_ref_2, T_ref_2, Delta_Cp_2)
    K_3 = K_(T, Delta_H_v_ref_3, T_ref_3, Delta_Cp_3)
    
    return (K_1*K_2)/(1 + K_1 + K_1*K_2 + K_1*K_2*K_3);

def alpha_3_(T, Delta_H_v_ref_1, Delta_H_v_ref_2, Delta_H_v_ref_3, T_ref_1, T_ref_2, T_ref_3, Delta_Cp_1, Delta_Cp_2, Delta_Cp_3):
    K_1 = K_(T, Delta_H_v_ref_1, T_ref_1, Delta_Cp_1)
    K_2 = K_(T, Delta_H_v_ref_2, T_ref_2, Delta_Cp_2)
    K_3 = K_(T, Delta_H_v_ref_3, T_ref_3, Delta_Cp_3)
    
    return (K_1*K_2*K_3)/(1 + K_1 + K_1*K_2 + K_1*K_2*K_3);
